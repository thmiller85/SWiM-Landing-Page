    <script>
        function updateScore() {
            const totalCheckboxes = document.querySelectorAll('input[type="checkbox"]').length;
            const checkedBoxes = document.querySelectorAll('input[type="checkbox"]:checked').length;
            
            const scoreDisplay = document.getElementById('scoreDisplay');
            const progressElement = document.getElementById('overallProgress');
            const progressFill = document.getElementById('progressFill');
            const progressText = document.getElementById('progressText');
            
            let priorityLevel = '';
            let priorityClass = '';
            
            if (checkedBoxes <= 5) {
                priorityLevel = 'Low Priority';
                priorityClass = 'priority-low';
            } else if (checkedBoxes <= 12) {
                priorityLevel = 'Moderate Priority';
                priorityClass = 'priority-medium';
            } else {
                priorityLevel = 'High Priority - Immediate Action Recommended';
                priorityClass = 'priority-high';
            }
            
            scoreDisplay.innerHTML = `
                Automation Priority: ${priorityLevel}<br>
                <small>Issues Identified: ${checkedBoxes} out of ${totalCheckboxes}</small>
            `;
            scoreDisplay.className = `score-display ${priorityClass}`;
            
            // Update progress bar
            const progressPercentage = (checkedBoxes / totalCheckboxes) * 100;
            progressFill.style.width = `${progressPercentage}%`;
            progressText.textContent = `Assessment Progress: ${checkedBoxes}/${totalCheckboxes} items checked (${Math.round(progressPercentage)}%)`;
            progressElement.style.display = 'block';
        }
        
        function calculatePriority(input) {
            const row = input.closest('tr');
            const inputs = row.querySelectorAll('input[type="number"]');
            const scoreSpan = row.querySelector('.priority-score');
            
            let total = 0;
            inputs.forEach(inp => {
                if (inp.value) total += parseInt(inp.value);
            });
            
            scoreSpan.textContent = total;
            
            // Highlight high-priority processes
            if (total >= 12) {
                scoreSpan.style.background = '#e74c3c';
                scoreSpan.style.color = 'white';
                scoreSpan.style.padding = '5px';
                scoreSpan.style.borderRadius = '3px';
            } else {
                scoreSpan.style.background = '';
                scoreSpan.style.color = '';
                scoreSpan.style.padding = '';
            }
        }
        
        function generateReport() {
            const checkedBoxes = document.querySelectorAll('input[type="checkbox"]:checked');
            const reportDiv = document.getElementById('assessmentReport');
            const reportContent = document.getElementById('reportContent');
            
            let report = `
                <div class="results-summary">
                    <h4>🎯 Assessment Summary</h4>
                    <p><strong>Issues Identified:</strong> ${checkedBoxes.length}</p>
                    <p><strong>Recommendation:</strong> ${getRecommendation(checkedBoxes.length)}</p>
                </div>
                
                <h4>📋 Issues Found:</h4>
                <ul>
            `;
            
            checkedBoxes.forEach(checkbox => {
                const label = document.querySelector(`label[for="${checkbox.id}"]`);
                report += `<li>${label.textContent}</li>`;
            });
            
            report += `</ul>
                
                <h4>🚀 Next Steps:</h4>
                <ol>
                    <li>Focus on processes with priority scores of 12+</li>
                    <li>Document current workflows for top 3 priority areas</li>
                    <li>Research automation solutions</li>
                    <li>Calculate ROI potential</li>
                    <li>Plan pilot implementation</li>
                </ol>
            `;
            
            reportContent.innerHTML = report;
            reportDiv.style.display = 'block';
            reportDiv.scrollIntoView({ behavior: 'smooth' });
        }
        
        function getRecommendation(score) {
            if (score <= 5) return 'Low automation priority. Monitor for future needs.';
            if (score <= 12) return 'Moderate automation opportunity. Start with highest-impact processes.';
            return 'High automation priority. Immediate action recommended to remain competitive.';
        }
        
        function saveProgress() {
            const formData = new FormData();
            const checkboxes = document.querySelectorAll('input[type="checkbox"]:checked');
            const inputs = document.querySelectorAll('input[type="number"], input[type="text"]');
            
            let data = { checkboxes: [], inputs: [] };
            
            checkboxes.forEach(cb => data.checkboxes.push(cb.id));
            inputs.forEach(inp => {
                if (inp.value) data.inputs.push({ id: inp.id || inp.name, value: inp.value });
            });
            
            localStorage.setItem('automationChecklist', JSON.stringify(data));
            alert('✅ Progress saved! You can return anytime to continue your assessment.');
        }
        
        function resetForm() {
            if (confirm('Are you sure you want to reset all data?')) {
                document.querySelectorAll('input').forEach(input => {
                    if (input.type === 'checkbox') input.checked = false;
                    else input.value = '';
                });
                document.querySelectorAll('.priority-score').forEach(span => span.textContent = '0');
                document.getElementById('scoreDisplay').innerHTML = 'Complete the assessment above to see your automation priority level';
                document.getElementById('assessmentReport').style.display = 'none';
                localStorage.removeItem('automationChecklist');
            }
        }
        
        // Load saved progress on page load
        window.onload = function() {
            const saved = localStorage.getItem('automationChecklist');
            if (saved) {
                const data = JSON.parse(saved);
                data.checkboxes.forEach(id => {
                    const checkbox = document.getElementById(id);
                    if (checkbox) checkbox.checked = true;
                });
                updateScore();
            }
        };
    </script>
